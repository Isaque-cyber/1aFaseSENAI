// Função para criar um card function criarCard(conteudo) { const card = document.createElement('div'); card.classList.add('Card'); // Adiciona uma classe para estilização card.textContent = conteudo; // Define o conteúdo do card return card; } // Recupera os dados do localStorage const dados = localStorage.getItem('habitos'); // Verifica se existem dados if (dados) { // Converte os dados de volta para um array (se estiverem armazenados como JSON) const dadosArray = JSON.parse(dados); // Obtém o elemento pai onde os cards serão adicionados const container = document.getElementById('container-cards'); // Itera sobre os dados e cria um card para cada item dadosArray.forEach(item => { const card = criarCard(item); container.appendChild(card); }); } else { // Se não houver dados, exibe uma mensagem const container = document.getElementById('container-cards'); container.textContent = 'Nenhum dado encontrado no localStorage.'; }
